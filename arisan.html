🔐 Pasang Firebase Rules
Copy & paste rules ini di Firebase Console → Firestore → Rules untuk membatasi akses orang luar:
rules_version = '2';
service cloud.firestore {
  match /databases/{db}/documents {
   match /config/{doc} {
    allow read, write: if false;
   }
   match /arisan/{id} {
    allow read: if true;
    allow write: if request.auth != null;
   }
  }
}
